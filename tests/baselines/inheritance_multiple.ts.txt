type PersonLikeBase = ThingBase & {
    "height"?: Number | Number[];
};
export type PersonLike = {
    "@type": "PersonLike";
} & PersonLikeBase;

type ThingBase = {
    /** IRI identifying the canonical address of this object. */
    "@id"?: string;
    "name"?: Text | Text[];
};
export type Thing = ({
    "@type": "Thing";
} & ThingBase) | (PersonLike | Vehicle);

type VehicleBase = ThingBase & {
    "doors"?: Number | Number[];
};
export type Vehicle = {
    "@type": "Vehicle";
} & VehicleBase;
